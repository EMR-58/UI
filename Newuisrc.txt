local function safeSetClipboard(text)
    local ok = pcall(setclipboard, text)
    if not ok then pcall(function() clipboard.set(text) end) end
end

local function createParticles(parent, count, color)
    if not parent then return end
    for _ = 1, count do
        local p = Instance.new("Frame")
        p.Size = UDim2.new(0, math.random(3,8), 0, math.random(3,8))
        p.Position = UDim2.new(math.random()*0.8+0.1, 0, math.random()*0.8+0.1, 0)
        p.BackgroundColor3 = color
        p.BorderSizePixel = 0
        p.ZIndex = 9999998
        p.Parent = parent
        local c = Instance.new("UICorner"); c.CornerRadius = UDim.new(1,0); c.Parent = p
        local t = TweenService:Create(p, TweenInfo.new(math.random(10,20)/10, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
            Position = UDim2.new(p.Position.X.Scale, math.random(-50,50), p.Position.Y.Scale-0.3, math.random(-50,50)),
            BackgroundTransparency = 1,
            Size = UDim2.new(0,0,0,0)
        })
        t:Play()
        t.Completed:Connect(function() p:Destroy() end)
    end
end

local function makeMainFrame(parent, w, h, name)
    local f = Instance.new("Frame")
    f.Name = name
    f.Size = UDim2.new(0, w, 0, h)
    f.Position = UDim2.new(0.5, 0, 0.45, 0)
    f.AnchorPoint = Vector2.new(0.5, 0.5)
    f.BackgroundColor3 = Color3.fromRGB(12, 12, 12)
    f.BorderSizePixel = 0
    f.ClipsDescendants = true
    f.Parent = parent

    local c = Instance.new("UICorner"); c.CornerRadius = UDim.new(0, 12); c.Parent = f
    local s = Instance.new("UIStroke")
    s.Thickness = 2.5
    s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    s.Color = Color3.fromRGB(255, 40, 40)
    s.Parent = f

    local g = Instance.new("UIGradient")
    g.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0, Color3.fromRGB(255, 40, 40)),
        ColorSequenceKeypoint.new(0.5, Color3.fromRGB(150, 0, 0)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(255, 40, 40))
    })
    g.Rotation = 45
    g.Parent = s

    task.spawn(function()
        while f.Parent do
            TweenService:Create(g, TweenInfo.new(3, Enum.EasingStyle.Linear), {Rotation = g.Rotation + 360}):Play()
            task.wait(3)
        end
    end)

    local stars = Instance.new("Frame")
    stars.Size = UDim2.new(1.5,0,1.5,0)
    stars.Position = UDim2.new(0,0,0,0)
    stars.BackgroundTransparency = 1
    stars.Parent = f
    for i=1, 35 do
        local st = Instance.new("Frame")
        st.Size = UDim2.new(0, math.random(1,2), 0, math.random(1,2))
        st.Position = UDim2.new(math.random(),0,math.random(),0)
        st.BackgroundColor3 = Color3.fromRGB(255,255,255)
        st.BackgroundTransparency = 0.5
        st.BorderSizePixel = 0
        st.Parent = stars
        Instance.new("UICorner").CornerRadius = UDim.new(1,0)
    end

    return f
end

local function makePanel(parent, xS, xO, wS, wO)
    local p = Instance.new("Frame")
    p.Position = UDim2.new(xS, xO, 0, 10)
    p.Size = UDim2.new(wS, wO, 1, -20)
    p.BackgroundTransparency = 1
    p.Parent = parent
    return p
end

local function hover(b)
    b.MouseEnter:Connect(function() TweenService:Create(b, TweenInfo.new(0.3), {BackgroundColor3 = Color3.fromRGB(45,45,45)}):Play() end)
    b.MouseLeave:Connect(function() TweenService:Create(b, TweenInfo.new(0.3), {BackgroundColor3 = Color3.fromRGB(30,30,30)}):Play() end)
end
