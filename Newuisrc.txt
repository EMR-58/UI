-- Neon Lazer Efektleri Oluşturucu
local function createLaser()
    local laser = Instance.new("Frame")
    laser.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    laser.BorderSizePixel = 0
    
    local isHorizontal = math.random() > 0.5
    if isHorizontal then
        laser.Size = UDim2.new(2, 0, 0, math.random(1, 3))
        laser.Position = UDim2.new(-0.5, 0, math.random(0, 100)/100, 0)
        laser.Rotation = math.random(-25, 25)
    else
        laser.Size = UDim2.new(0, math.random(1, 3), 2, 0)
        laser.Position = UDim2.new(math.random(0, 100)/100, 0, -0.5, 0)
        laser.Rotation = math.random(-25, 25)
    end
    
    -- Neon Parlaması
    local glow = Instance.new("UIStroke")
    glow.Color = Color3.fromRGB(255, 30, 30)
    glow.Thickness = math.random(3, 8)
    glow.Transparency = 0.5
    glow.Parent = laser
    
    laser.ZIndex = 3
    laser.Parent = EffectContainer
    
    local duration = math.random(4, 10)
    local targetPos
    if isHorizontal then
        targetPos = UDim2.new(laser.Position.X.Scale, 0, laser.Position.Y.Scale + (math.random(-50, 50)/100), 0)
    else
        targetPos = UDim2.new(laser.Position.X.Scale + (math.random(-50, 50)/100), 0, laser.Position.Y.Scale, 0)
    end

    local tween = TweenService:Create(laser, TweenInfo.new(duration, Enum.EasingStyle.Linear), {
        Position = targetPos,
        BackgroundTransparency = 1
    })
    
    TweenService:Create(glow, TweenInfo.new(duration, Enum.EasingStyle.Linear), {Transparency = 1}):Play()
    tween:Play()
    tween.Completed:Connect(function() laser:Destroy() end)
end

    while task.wait(0.15) do
        if not Background.Parent then break end
        createLaser()
    end
end)

local function createGlassPanel(name, size, pos)
    local panel = Instance.new("Frame")
    panel.Name = name
    panel.Size = size
    panel.Position = pos
    panel.AnchorPoint = Vector2.new(0.5, 0.5)
    panel.BackgroundColor3 = Color3.fromRGB(15, 0, 5)
    panel.BackgroundTransparency = 0.3
    panel.BorderSizePixel = 0
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 12)
    corner.Parent = panel
    
    local stroke = Instance.new("UIStroke")
    stroke.Color = Color3.fromRGB(255, 0, 0)
    stroke.Thickness = 1.5
    stroke.Transparency = 0.2
    stroke.Parent = panel

    local glow = Instance.new("UIGradient")
    glow.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0, Color3.fromRGB(255, 40, 40)),
        ColorSequenceKeypoint.new(0.5, Color3.fromRGB(50, 0, 0)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(255, 40, 40))
    })
    glow.Rotation = 45
    glow.Parent = stroke
    
    task.spawn(function()
        while panel.Parent do
            TweenService:Create(glow, TweenInfo.new(3, Enum.EasingStyle.Linear), {Rotation = glow.Rotation + 360}):Play()
            task.wait(3)
        end
    end)
    
    return panel
end

local function applyHoverEffect(button)
    button.MouseEnter:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.2), {BackgroundTransparency = 0.1, Size = UDim2.new(button.Size.X.Scale, button.Size.X.Offset + 4, button.Size.Y.Scale, button.Size.Y.Offset + 2)}):Play()
    end)
    button.MouseLeave:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.2), {BackgroundTransparency = 0.4, Size = UDim2.new(button.Size.X.Scale, button.Size.X.Offset - 4, button.Size.Y.Scale, button.Size.Y.Offset - 2)}):Play()
    end)
end

local function playLoadingSequence(text, duration, callback)
    LoadingTitle.Text = text
    BarFill.Size = UDim2.new(0, 0, 1, 0)
    PercentText.Text = "0%"
    
    LoadingContainer.Position = UDim2.new(0.5, 0, 0.5, 50)
    LoadingContainer.GroupTransparency = 1
    LoadingContainer.Visible = true
    
    -- Yükleme ekranı giriş animasyonu
    TweenService:Create(LoadingContainer, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Position = UDim2.new(0.5, 0, 0.5, 0)}):Play()
    local fade = Instance.new("CanvasGroup") -- Transparency kontrolü için
    
    local fillTween = TweenService:Create(BarFill, TweenInfo.new(duration, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), {Size = UDim2.new(1, 0, 1, 0)})
    
    local connection
    connection = RunService.RenderStepped:Connect(function()
        local percent = math.floor(BarFill.Size.X.Scale * 100)
        PercentText.Text = tostring(percent) .. "%"
    end)
    
    fillTween:Play()
    fillTween.Completed:Wait()
    connection:Disconnect()
    PercentText.Text = "100%"
    
    task.wait(0.3)
    
    -- Yükleme ekranı çıkış animasyonu
    local outTween = TweenService:Create(LoadingContainer, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.In), {Position = UDim2.new(0.5, 0, 0.5, -50)})
    outTween:Play()
    outTween.Completed:Wait()
    LoadingContainer.Visible = false
    
    if callback then callback() end
end

                    if stroke:IsA("UIStroke") then
                        TweenService:Create(stroke, TweenInfo.new(1), {Transparency = 1}):Play()

                end
            end
