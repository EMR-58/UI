-- 1. Kütüphaneyi Linkten Çekiyoruz (Kendi linkini buraya koy)
local SynioxFusion = loadstring(game:HttpGet("https://raw.githubusercontent.com/EMR-58/UI/refs/heads/main/Elerium%20ui.txt"))()

-- 2. Ana Pencereyi Oluştur (Elerium Tasarımı & WindUI Fonksiyonları)
local Window = SynioxFusion:CreateWindow({
    Title = "Syniox Hub | Muscle Legends",
    Theme = "Crimson", -- Senin istediğin kan kırmızısı geçişler
    Size = UDim2.new(0, 680, 0, 460)
})

-- 3. Tabları (Sekmeleri) Oluştur
local MainTab = Window:Tab({Title = "Main"})
local FarmTab = Window:Tab({Title = "Farm"})
local StatsTab = Window:Tab({Title = "Stats"})

-- 4. Farm Sekmesine Özellikleri Ekle (Örn: Auto Weight)
-- Sen yukarıda Search'e "Auto Weight" yazınca buranın açılmasını istemiştin.
-- Sistem otomatik olarak bu başlığı tarar ve bulur.

FarmTab:Toggle({
    Title = "Auto Weight",
    Default = false,
    Callback = function(Value)
        _G.AutoWeight = Value
        while _G.AutoWeight do
            task.wait()
            -- Buraya Muscle Legends ağırlık kaldırma remote kodunu yazacaksın
            game:GetService("Players").LocalPlayer.muscleEvent:FireServer("reps")
        end
    end
})

FarmTab:Toggle({
    Title = "Auto Pushups",
    Default = false,
    Callback = function(Value)
        _G.AutoPushups = Value
        while _G.AutoPushups do
            task.wait()
            game:GetService("Players").LocalPlayer.muscleEvent:FireServer("pushups")
        end
    end
})

-- 5. Main Sekmesine Diğer Özellikleri Ekle
MainTab:Slider({
    Title = "WalkSpeed",
    Min = 16,
    Max = 250,
    Default = 16,
    Callback = function(Value)
        game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = Value
    end
})

MainTab:Button({
    Title = "Rebirth",
    Callback = function()
        game:GetService("ReplicatedStorage").rEvents.rebirthEvent:FireServer("rebirthRequest")
    end
})
