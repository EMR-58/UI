-- Gelişmiş Animasyon Fonksiyonu
local function Tween(instance, properties, duration, style, direction)
    local tInfo = TweenInfo.new(duration or 0.3, style or Enum.EasingStyle.Quint, direction or Enum.EasingDirection.Out)
    local tween = TweenService:Create(instance, tInfo, properties)
    tween:Play()
    return tween
end

-- Elerium Aktif Tab Parçacık Efekti (Beyaz Topuklar)
local function EmitParticles(parentFrame)
    local particleLoop = true
    task.spawn(function()
        while particleLoop do
            if not parentFrame or not parentFrame.Parent then break end
            if parentFrame.BackgroundTransparency == 1 then task.wait(0.1) continue end -- Sadece aktif tabda çalışır
            
            local particle = Instance.new("Frame")
            particle.Name = "GlowParticle"
            particle.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
            particle.BorderSizePixel = 0
            
            -- Rastgele boyut ve pozisyon
            local size = math.random(2, 5)
            particle.Size = UDim2.new(0, size, 0, size)
            particle.Position = UDim2.new(math.random(10, 90)/100, 0, 1, 0)
            particle.BackgroundTransparency = 0.4
            particle.ZIndex = parentFrame.ZIndex + 1
            
            local pCorner = Instance.new("UICorner")
            pCorner.CornerRadius = UDim.new(1, 0)
            pCorner.Parent = particle
            
            particle.Parent = parentFrame
            
            -- Yukarı doğru süzülme animasyonu
            local floatUp = TweenService:Create(particle, TweenInfo.new(math.random(15, 30)/10), {
                Position = UDim2.new(particle.Position.X.Scale, 0, 0, -10),
                BackgroundTransparency = 1
            })
            
            floatUp:Play()
            floatUp.Completed:Connect(function()
                particle:Destroy()
            end)
            
            task.wait(math.random(1, 4) / 10)
        end
    end)
    return function() particleLoop = false end
end

            local function UpdateSlider(input)
                local pos = math.clamp(input.Position.X - BarArea.AbsolutePosition.X, 0, BarArea.AbsoluteSize.X)
                local percent = pos / BarArea.AbsoluteSize.X
                Value = math.floor(Min + (Max - Min) * percent)
                ValueLabel.Text = tostring(Value)
                Tween(Fill, {Size = UDim2.new(percent, 0, 1, 0)}, 0.1)
                Callback(Value)
            end

            BarArea.InputBegan:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 then
                    dragging = true
                    Tween(UI_Glow, {Transparency = 0.5}, 0.2)
                    UpdateSlider(input)
                end
            end)

            UserInputService.InputEnded:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 then
                    dragging = false
                    Tween(UI_Glow, {Transparency = 1}, 0.2)
                end
            end)

            UserInputService.InputChanged:Connect(function(input)
                if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
                    UpdateSlider(input)
                end
            end)
        end

        return Elements
    end

    return GUI
end

return SynioxFusion
