local ScreenGui = Instance.new("ScreenGui")

local function CreateTween(instance, properties, duration)
    local tweenInfo = TweenInfo.new(duration or 0.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out)
    local tween = TweenService:Create(instance, tweenInfo, properties)
    tween:Play()
    return tween
end

    local StartPosition
    
    Topbar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            Dragging = true
            DragStart = input.Position
            StartPosition = MainFrame.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    Dragging = false
                end
            end)
        end
    end)
    
    Topbar.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            DragInput = input
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if input == DragInput and Dragging then
            local delta = input.Position - DragStart
            MainFrame.Position = UDim2.new(StartPosition.X.Scale, StartPosition.X.Offset + delta.X, StartPosition.Y.Scale, StartPosition.Y.Offset + delta.Y)
        end
    end)

    ContentArea.Name = "ContentArea"

    TitleLabel.Name = "TitleLabel"
    TitleLabel.Parent = ContentArea
    TitleLabel.BackgroundTransparency = 1
    TitleLabel.Position = UDim2.new(0, 20, 0, 15)
    TitleLabel.Size = UDim2.new(1, -40, 0, 30)

    TitleLabel.Text = WindowTitle
    TitleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)

    TitleLabel.TextXAlignment = Enum.TextXAlignment.Left

    Pages.Name = "Pages"

    Pages.BackgroundTransparency = 1
    Pages.Position = UDim2.new(0, 0, 0, 55)
    Pages.Size = UDim2.new(1, 0, 1, -55)
    
    SearchBox.Changed:Connect(function(prop)
        if prop == "Text" then
            local searchText = string.lower(SearchBox.Text)
            for _, tabBtn in pairs(TabContainer:GetChildren()) do
                if tabBtn:IsA("TextButton") then
                    if searchText == "" or string.find(string.lower(tabBtn.Text), searchText) then
                        tabBtn.Visible = true
                    else
                        tabBtn.Visible = false
                    end
                end
            end
        end
    end)
    
    local Window = {
        Tabs = {},
        CurrentTab = nil

    function Window:Tab(tabConfig)
        local tabTitle = tabConfig.Title or "Tab"
        
        local TabButton = Instance.new("TextButton")
        TabButton.Name = tabTitle
        TabButton.Parent = TabContainer
        TabButton.BackgroundColor3 = Color3.fromRGB(0, 120, 215)
        TabButton.BackgroundTransparency = 1
        TabButton.Size = UDim2.new(1, 0, 0, 32)
        TabButton.Font = Enum.Font.GothamMedium
        TabButton.Text = "   " .. tabTitle
        TabButton.TextColor3 = Color3.fromRGB(160, 160, 165)
        TabButton.TextSize = 13
        TabButton.TextXAlignment = Enum.TextXAlignment.Left
        TabButton.AutoButtonColor = false
        
        local TabButtonCorner = Instance.new("UICorner")
        TabButtonCorner.CornerRadius = UDim.new(0, 6)
        TabButtonCorner.Parent = TabButton
        
        local Page = Instance.new("ScrollingFrame")
        Page.Name = tabTitle
        Page.Parent = Pages
        Page.BackgroundTransparency = 1
        Page.Size = UDim2.new(1, -40, 1, -20)
        Page.Position = UDim2.new(0, 20, 0, 0)
        Page.ScrollBarThickness = 2
        Page.ScrollBarImageColor3 = Color3.fromRGB(60, 60, 65)
        Page.Visible = false
        Page.CanvasSize = UDim2.new(0, 0, 0, 0)
        
        local PageLayout = Instance.new("UIListLayout")
        PageLayout.Parent = Page
        PageLayout.SortOrder = Enum.SortOrder.LayoutOrder
        PageLayout.Padding = UDim.new(0, 6)
        
        PageLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
            Page.CanvasSize = UDim2.new(0, 0, 0, PageLayout.AbsoluteContentSize.Y + 10)
        end)
        
        TabButton.MouseButton1Click:Connect(function()
            for _, page in pairs(Pages:GetChildren()) do
                if page:IsA("ScrollingFrame") then
                    page.Visible = false
                end
            end
            for _, btn in pairs(TabContainer:GetChildren()) do
                if btn:IsA("TextButton") then
                    CreateTween(btn, {BackgroundTransparency = 1, TextColor3 = Color3.fromRGB(160, 160, 165)})
                end
            end
            Page.Visible = true
            CreateTween(TabButton, {BackgroundTransparency = 0.8, TextColor3 = Color3.fromRGB(255, 255, 255)})
        end)
        
        local TabData = {}
        
        function TabData:Button(btnConfig)
            local btnTitle = btnConfig.Title or "Button"
            local btnDesc = btnConfig.Desc or ""
            local callback = btnConfig.Callback or function() end
            
            local Button = Instance.new("TextButton")
            Button.Name = btnTitle
            Button.Parent = Page
            Button.BackgroundColor3 = Color3.fromRGB(25, 25, 28)
            Button.Size = UDim2.new(1, 0, 0, btnDesc == "" and 40 or 50)
            Button.Font = Enum.Font.Gotham
            Button.Text = ""
            Button.AutoButtonColor = false
            
            local BtnCorner = Instance.new("UICorner")
            BtnCorner.CornerRadius = UDim.new(0, 6)
            BtnCorner.Parent = Button
            
            local BtnStroke = Instance.new("UIStroke")
            BtnStroke.Color = Color3.fromRGB(45, 45, 50)
            BtnStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
            BtnStroke.Parent = Button
            
            local TitleLabel = Instance.new("TextLabel")
            TitleLabel.Parent = Button
            TitleLabel.BackgroundTransparency = 1
            TitleLabel.Position = UDim2.new(0, 15, 0, btnDesc == "" and 0 or 8)
            TitleLabel.Size = UDim2.new(1, -30, 0, btnDesc == "" and 40 or 16)
            TitleLabel.Font = Enum.Font.GothamMedium
            TitleLabel.Text = btnTitle
            TitleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
            TitleLabel.TextSize = 14
            TitleLabel.TextXAlignment = Enum.TextXAlignment.Left
            
            if btnDesc ~= "" then
                local DescLabel = Instance.new("TextLabel")
                DescLabel.Parent = Button
                DescLabel.BackgroundTransparency = 1
                DescLabel.Position = UDim2.new(0, 15, 0, 26)
                DescLabel.Size = UDim2.new(1, -30, 0, 14)
                DescLabel.Font = Enum.Font.Gotham
                DescLabel.Text = btnDesc
                DescLabel.TextColor3 = Color3.fromRGB(150, 150, 150)
                DescLabel.TextSize = 12
                DescLabel.TextXAlignment = Enum.TextXAlignment.Left
            end
            
            Button.MouseEnter:Connect(function()
                CreateTween(Button, {BackgroundColor3 = Color3.fromRGB(32, 32, 36)})
            end)
            Button.MouseLeave:Connect(function()
                CreateTween(Button, {BackgroundColor3 = Color3.fromRGB(25, 25, 28)})
            end)
            Button.MouseButton1Click:Connect(function()
                CreateTween(Button, {BackgroundColor3 = Color3.fromRGB(45, 45, 50)}, 0.1)
                task.delay(0.1, function()
                    CreateTween(Button, {BackgroundColor3 = Color3.fromRGB(32, 32, 36)}, 0.1)
                end)
                callback()
            end)
        end
        
        function TabData:Toggle(toggleConfig)
            local togTitle = toggleConfig.Title or "Toggle"
            local togDesc = toggleConfig.Desc or ""
            local default = toggleConfig.Default or false
            local callback = toggleConfig.Callback or function() end
            
            local ToggleVal = default
            
            local ToggleFrame = Instance.new("TextButton")
            ToggleFrame.Name = togTitle
            ToggleFrame.Parent = Page
            ToggleFrame.BackgroundColor3 = Color3.fromRGB(25, 25, 28)
            ToggleFrame.Size = UDim2.new(1, 0, 0, togDesc == "" and 40 or 50)
            ToggleFrame.Font = Enum.Font.Gotham
            ToggleFrame.Text = ""
            ToggleFrame.AutoButtonColor = false
            
            local TogCorner = Instance.new("UICorner")
            TogCorner.CornerRadius = UDim.new(0, 6)
            TogCorner.Parent = ToggleFrame
            
            local TogStroke = Instance.new("UIStroke")
            TogStroke.Color = Color3.fromRGB(45, 45, 50)
            TogStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
            TogStroke.Parent = ToggleFrame
            
            local TitleLabel = Instance.new("TextLabel")
            TitleLabel.Parent = ToggleFrame
            TitleLabel.BackgroundTransparency = 1
            TitleLabel.Position = UDim2.new(0, 15, 0, togDesc == "" and 0 or 8)
            TitleLabel.Size = UDim2.new(1, -80, 0, togDesc == "" and 40 or 16)
            TitleLabel.Font = Enum.Font.GothamMedium
            TitleLabel.Text = togTitle
            TitleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
            TitleLabel.TextSize = 14
            TitleLabel.TextXAlignment = Enum.TextXAlignment.Left
            
            if togDesc ~= "" then
                local DescLabel = Instance.new("TextLabel")
                DescLabel.Parent = ToggleFrame
                DescLabel.BackgroundTransparency = 1
                DescLabel.Position = UDim2.new(0, 15, 0, 26)
                DescLabel.Size = UDim2.new(1, -80, 0, 14)
                DescLabel.Font = Enum.Font.Gotham
                DescLabel.Text = togDesc
                DescLabel.TextColor3 = Color3.fromRGB(150, 150, 150)
                DescLabel.TextSize = 12
                DescLabel.TextXAlignment = Enum.TextXAlignment.Left
            end
            
            local SwitchBg = Instance.new("Frame")
            SwitchBg.Parent = ToggleFrame
            SwitchBg.BackgroundColor3 = ToggleVal and Color3.fromRGB(0, 120, 215) or Color3.fromRGB(40, 40, 45)
            SwitchBg.Position = UDim2.new(1, -45, 0.5, -10)
            SwitchBg.Size = UDim2.new(0, 34, 0, 20)
            
            local SwitchCorner = Instance.new("UICorner")
            SwitchCorner.CornerRadius = UDim.new(1, 0)
            SwitchCorner.Parent = SwitchBg
            
            local SwitchDot = Instance.new("Frame")
            SwitchDot.Parent = SwitchBg
            SwitchDot.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
            SwitchDot.Position = ToggleVal and UDim2.new(0, 16, 0.5, -7) or UDim2.new(0, 2, 0.5, -7)
            SwitchDot.Size = UDim2.new(0, 14, 0, 14)
            
            local DotCorner = Instance.new("UICorner")
            DotCorner.CornerRadius = UDim.new(1, 0)
            DotCorner.Parent = SwitchDot
            
            ToggleFrame.MouseButton1Click:Connect(function()
                ToggleVal = not ToggleVal
                if ToggleVal then
                    CreateTween(SwitchBg, {BackgroundColor3 = Color3.fromRGB(0, 120, 215)})
                    CreateTween(SwitchDot, {Position = UDim2.new(0, 16, 0.5, -7)})
                else
                    CreateTween(SwitchBg, {BackgroundColor3 = Color3.fromRGB(40, 40, 45)})
                    CreateTween(SwitchDot, {Position = UDim2.new(0, 2, 0.5, -7)})
                end
                callback(ToggleVal)
            end)
        end
        
        function TabData:Slider(sliderConfig)
            local sTitle = sliderConfig.Title or "Slider"
            local sMin = sliderConfig.Min or 0
            local sMax = sliderConfig.Max or 100
            local sDef = sliderConfig.Default or sMin
            local callback = sliderConfig.Callback or function() end
            
            local SliderVal = sDef
            
            local SliderFrame = Instance.new("Frame")
            SliderFrame.Name = sTitle
            SliderFrame.Parent = Page
            SliderFrame.BackgroundColor3 = Color3.fromRGB(25, 25, 28)
            SliderFrame.Size = UDim2.new(1, 0, 0, 55)
            
            local SCorner = Instance.new("UICorner")
            SCorner.CornerRadius = UDim.new(0, 6)
            SCorner.Parent = SliderFrame
            
            local SStroke = Instance.new("UIStroke")
            SStroke.Color = Color3.fromRGB(45, 45, 50)
            SStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
            SStroke.Parent = SliderFrame
            
            local TitleLabel = Instance.new("TextLabel")
            TitleLabel.Parent = SliderFrame
            TitleLabel.BackgroundTransparency = 1
            TitleLabel.Position = UDim2.new(0, 15, 0, 10)
            TitleLabel.Size = UDim2.new(1, -60, 0, 16)
            TitleLabel.Font = Enum.Font.GothamMedium
            TitleLabel.Text = sTitle
            TitleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
            TitleLabel.TextSize = 14

TitleLabel.TextXAlignment = Enum.TextXAlignment.Left
            
            local ValLabel = Instance.new("TextLabel")
            ValLabel.Parent = SliderFrame
            ValLabel.BackgroundTransparency = 1
            ValLabel.Position = UDim2.new(1, -45, 0, 10)
            ValLabel.Size = UDim2.new(0, 30, 0, 16)
            ValLabel.Font = Enum.Font.Gotham
            ValLabel.Text = tostring(sDef)
            ValLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
            ValLabel.TextSize = 13
            ValLabel.TextXAlignment = Enum.TextXAlignment.Right
            
            local SliderBg = Instance.new("TextButton")
            SliderBg.Parent = SliderFrame
            SliderBg.BackgroundColor3 = Color3.fromRGB(40, 40, 45)
            SliderBg.Position = UDim2.new(0, 15, 0, 35)
            SliderBg.Size = UDim2.new(1, -30, 0, 6)
            SliderBg.Text = ""
            SliderBg.AutoButtonColor = false
            
            local BgCorner = Instance.new("UICorner")
            BgCorner.CornerRadius = UDim.new(1, 0)
            BgCorner.Parent = SliderBg
            
            local SliderFill = Instance.new("Frame")
            SliderFill.Parent = SliderBg
            SliderFill.BackgroundColor3 = Color3.fromRGB(0, 120, 215)
            SliderFill.Size = UDim2.new((sDef - sMin) / (sMax - sMin), 0, 1, 0)
            
            local FillCorner = Instance.new("UICorner")
            FillCorner.CornerRadius = UDim.new(1, 0)
            FillCorner.Parent = SliderFill
            
            local SliderDragging = false
            
            local function UpdateSlider(input)
                local pos = math.clamp(input.Position.X - SliderBg.AbsolutePosition.X, 0, SliderBg.AbsoluteSize.X)
                local percent = pos / SliderBg.AbsoluteSize.X
                SliderFill.Size = UDim2.new(percent, 0, 1, 0)
                SliderVal = math.floor(sMin + (sMax - sMin) * percent)
                ValLabel.Text = tostring(SliderVal)
                callback(SliderVal)
            end
            
            SliderBg.InputBegan:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                    SliderDragging = true
                    UpdateSlider(input)
                end
            end)
            
            UserInputService.InputEnded:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                    SliderDragging = false
                end
            end)
            
            UserInputService.InputChanged:Connect(function(input)
                if SliderDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                    UpdateSlider(input)
                end
            end)
        end
        
        return TabData
    end
    
    return Window
end

return Library
